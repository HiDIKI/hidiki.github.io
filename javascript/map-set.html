    const secrets = new WeakMap();
    return class {
        setSecret(secret){
            secrets.set(this, secret);
        }
        getSecret(){
            return secrets.get(this);
        }

b.setSecret('secret B');

a.getSecret(); // secret A
b.getSecret(); // secret B
